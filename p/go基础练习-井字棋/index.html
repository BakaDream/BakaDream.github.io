package main

import (
	"fmt"
	"os"
	"os/exec"
)

var Board = [3][3]int{}
var X int
var Y int
var Counter int

func ClearDisplay() {
	cmd := exec.Command("cmd", "/c", "cls")
	cmd.Stdout = os.Stdout
	cmd.Run()
}
func Judgement() int {
	//当前行判断
	if Board[Y][0]+Board[Y][1]+Board[Y][2] == 3 || Board[Y][0]+Board[Y][1]+Board[Y][2] == -3 {
		return 1
	}
	//当前列判断
	if Board[0][X]+Board[1][X]+Board[2][X] == 3 || Board[0][X]+Board[1][X]+Board[2][X] == -3 {
		return 1
	}
	//两个斜线判断
	if Board[0][0]+Board[1][1]+Board[2][2] == 3 || Board[2][0]+Board[1][1]+Board[0][2] == 3 {
		return 1
	}
	if Board[0][0]+Board[1][1]+Board[2][2] == -3 || Board[2][0]+Board[1][1]+Board[0][2] == -3 {
		return 1
	}
	//平局
	if Counter == 9 {
		return 2
	}
	return 0
}
func PrintBoard() {
	fmt.Println("已进行", Counter, "回合")
	if Counter%2 == 0 {
		println("请o落子")
	} else {
		println("请x落子")
	}
	fmt.Println("Y")
	//控制列
	for y := 2; y > -1; y-- {
		fmt.Printf("%d ", y+1)
		//打印出当前行
		for x := 0; x < 3; x++ {
			switch Board[y][x] {
			case -1:
				fmt.Printf(" o ")
			case 0:
				fmt.Printf(" _ ")
			case 1:
				fmt.Printf(" x ")
			}
		}
		//打印完Y行后
		fmt.Printf("\n")
	}
	fmt.Println("   1  2  3  X")
}
func PlayerMove() {
	for {
		fmt.Printf("请输入X轴坐标:")
		fmt.Scanln(&X)
		fmt.Printf("请输入Y轴坐标:")
		fmt.Scanln(&Y)
		X = X - 1
		Y = Y - 1
		//用户输入坐标合法判断
		if X > -1 && X < 3 && Y > -1 && Y < 3 {
			if Board[Y][X] == 0 {
				if Counter%2 == 0 {
					Board[Y][X] = -1
				} else {
					Board[Y][X] = 1
				}
				break
			} else {
				println("位置已占用，请重新输入!")
			}
		} else {
			println("输入不合法，请重新输入!")
		}
	}
}
func main() {
	var judge = 0
	for {
		PrintBoard()
		PlayerMove()
		Counter++
		judge = Judgement()
		if judge != 0 {
			if judge == 2 {
				fmt.Println("平局！")
			} else {
				if (Counter-1)%2 == 0 {
					fmt.Println("恭喜 o 方胜利！")
				} else {
					fmt.Println("恭喜 x 方胜利！")
				}
			}
			fmt.Println("共进行", Counter, "回合")
			break
		}
		ClearDisplay()
	}
}
